#include <WiFi.h>
#include <WiFiClientSecure.h>
#include <PubSubClient.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <WiFiManager.h> 

// HiveMQ Credentials
const char* mqtt_server = "16e51255d95244c2b069b92cf77ebf81.s1.eu.hivemq.cloud";
const char* mqtt_user = "RheinTigle";
const char* mqtt_pass = "052105@Rhein";

// MQTT Topics
const char* topic_data = "hfire/house1/data";
const char* topic_status = "hfire/house1/status";
const char* topic_config = "hfire/house1/config"; 

// Pins & Thresholds
const int MQ2_PIN = 34;
const int BUZZER_PIN = 25;
const int SAFE_LIMIT = 450;
const int DANGER_LIMIT = 1500;

WiFiClientSecure espClient;
PubSubClient client(espClient);
LiquidCrystal_I2C lcd(0x27, 16, 2);

String deviceMac = "";

void mqttCallback(char* topic, byte* payload, unsigned int length) {
  String message;
  for (int i = 0; i < length; i++) message += (char)payload[i];

  if (String(topic) == topic_config && message == "RESET_WIFI") {
    lcd.clear();
    lcd.print("Resetting WiFi...");
    delay(2000);
    WiFiManager wm;
    wm.resetSettings(); 
    ESP.restart();      
  }
}

void configModeCallback (WiFiManager *myWiFiManager) {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("WiFi Setup Mode");
  lcd.setCursor(0, 1);
  lcd.print("AP: H-Fire-Setup");
}

void setup_wifi() {
  WiFiManager wm;
  wm.setAPCallback(configModeCallback);
  wm.setConnectTimeout(30);
  wm.setConfigPortalTimeout(180);

  if (!wm.autoConnect("H-Fire-Setup")) {
    lcd.clear();
    lcd.print("Setup Timeout");
    delay(3000);
    ESP.restart();
  }

  deviceMac = WiFi.macAddress();
  espClient.setInsecure(); 
}

void reconnect() {
  while (!client.connected()) {
    // 1. Check if WiFi is still alive
    if (WiFi.status() != WL_CONNECTED) {
      lcd.setCursor(0, 1);
      lcd.print("WiFi Lost!      "); // Clear line with spaces
      delay(2000);
      ESP.restart(); // Trigger Hotspot via restart
    }

    // 2. Try MQTT Connection
    Serial.print("Attempting MQTT connection...");
    if (client.connect(deviceMac.c_str(), mqtt_user, mqtt_pass)) {
      Serial.println("connected");
      client.subscribe(topic_config);
      client.publish(topic_status, "System Online");
      
      // Clear the "Retry" message from line 2
      lcd.setCursor(0, 1);
      lcd.print("Connected!      ");
      delay(1000);
    } else {
      // Show Retry on Line 2, but keep Line 1 (PPM) visible
      lcd.setCursor(0, 1);
      lcd.print("MQTT Retry...   "); // 16 characters to clear line
      delay(5000);
    }
  }
}

void setup() {
  Serial.begin(115200);
  pinMode(BUZZER_PIN, OUTPUT);
  lcd.init();
  lcd.backlight();
  
  lcd.setCursor(0,0);
  lcd.print("H-Fire Booting..");
  setup_wifi();
  
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("ID: ");
  lcd.print(deviceMac.substring(9));
  lcd.setCursor(0, 1);
  lcd.print("System Ready");
  delay(2000);
  
  client.setServer(mqtt_server, 8883);
  client.setCallback(mqttCallback); 
}

void loop() {
  // If WiFi breaks, restart immediately to open hotspot
  if (WiFi.status() != WL_CONNECTED) {
    lcd.setCursor(0, 1);
    lcd.print("WiFi Lost!      ");
    delay(2000);
    ESP.restart();
  }

  if (!client.connected()) {
    reconnect();
  }
  client.loop();

  int ppm = analogRead(MQ2_PIN);
  String statusMsg;

  if (ppm <= SAFE_LIMIT) {
    statusMsg = "SAFE            ";
    digitalWrite(BUZZER_PIN, LOW);
  } else if (ppm > SAFE_LIMIT && ppm <= DANGER_LIMIT) {
    statusMsg = "WARNING: SMOKE  ";
    digitalWrite(BUZZER_PIN, HIGH); 
    delay(100); 
    digitalWrite(BUZZER_PIN, LOW);
  } else {
    statusMsg = "CRITICAL: FIRE  ";
    digitalWrite(BUZZER_PIN, HIGH);
  }

  // Row 1: PPM (Always visible)
  lcd.setCursor(0, 0);
  lcd.print("PPM: "); 
  lcd.print(ppm);
  lcd.print("     "); // Clear trailing numbers

  // Row 2: Status
  lcd.setCursor(0, 1); 
  lcd.print(statusMsg);

  // Data Publish
  String dataPayload = "{\"mac\":\"" + deviceMac + "\", \"ppm\":" + String(ppm) + "}";
  client.publish(topic_data, dataPayload.c_str());

  delay(2000);
}
