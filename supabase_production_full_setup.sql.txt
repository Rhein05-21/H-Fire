-- ==========================================

-- 1. TABLE DEFINITIONS
-- ==========================================
CREATE TABLE IF NOT EXISTS public.profiles (
  id text NOT NULL,
  name text NOT NULL,
  community text,
  updated_at timestamp without time zone DEFAULT now(),
  latitude double precision,
  longitude double precision,
  is_admin boolean DEFAULT false,
  CONSTRAINT profiles_pkey PRIMARY KEY (id)
);

CREATE TABLE IF NOT EXISTS public.devices (
  mac text NOT NULL,
  house_name text NOT NULL,
  label text NOT NULL,
  profile_id text REFERENCES public.profiles(id),
  created_at timestamp with time zone DEFAULT now(),
  last_seen timestamp with time zone,
  community text,
  CONSTRAINT devices_pkey PRIMARY KEY (mac)
);

CREATE TABLE IF NOT EXISTS public.gas_logs (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  created_at timestamp with time zone DEFAULT now(),
  device_mac text REFERENCES public.devices(mac),
  ppm_level integer,
  status text,
  profile_id text REFERENCES public.profiles(id)
);

CREATE TABLE IF NOT EXISTS public.incidents (
  id bigint GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  device_mac text NOT NULL REFERENCES public.devices(mac),
  status text NOT NULL DEFAULT 'Active',
  start_time timestamp with time zone DEFAULT now(),
  end_time timestamp with time zone,
  resolved_by text REFERENCES public.profiles(id),
  notes text,
  ppm_at_trigger integer,
  alert_type text,
  profile_id text REFERENCES public.profiles(id)
);

-- 2. SECURITY LAYER
-- ==========================================
ALTER TABLE public.profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.devices ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.gas_logs ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.incidents ENABLE ROW LEVEL SECURITY;

-- 3. ADMIN HELPER FUNCTION (FIXED WITH TYPE CAST)
-- ==========================================
CREATE OR REPLACE FUNCTION public.is_admin()
RETURNS BOOLEAN AS $$
  -- Added ::text to convert UUID to TEXT for comparison
  SELECT COALESCE(is_admin, FALSE) FROM public.profiles WHERE id = auth.uid()::text;
$$ LANGUAGE sql SECURITY DEFINER;

-- 4. ACCESS CONTROL POLICIES (FIXED WITH TYPE CASTS)
-- ==========================================

-- PROFILES
DROP POLICY IF EXISTS "Profiles: Viewable by owner/admin" ON public.profiles;
CREATE POLICY "Profiles: Viewable by owner/admin" ON public.profiles
FOR SELECT USING (auth.uid()::text = id OR is_admin());

-- DEVICES
DROP POLICY IF EXISTS "Devices: Resident owned or Admin all" ON public.devices;
CREATE POLICY "Devices: Resident owned or Admin all" ON public.devices
FOR SELECT USING (is_admin() OR profile_id = auth.uid()::text);

-- GAS LOGS
DROP POLICY IF EXISTS "Gas Logs: Resident owned or Admin all" ON public.gas_logs;
CREATE POLICY "Gas Logs: Resident owned or Admin all" ON public.gas_logs
FOR SELECT USING (is_admin() OR profile_id = auth.uid()::text);

-- INCIDENTS
DROP POLICY IF EXISTS "Incidents: Resident owned or Admin all" ON public.incidents;
CREATE POLICY "Incidents: Resident owned or Admin all" ON public.incidents
FOR SELECT USING (is_admin() OR profile_id = auth.uid()::text);

-- SYSTEM INSERTS
DROP POLICY IF EXISTS "System: Allow Bridge Inserts" ON public.gas_logs;
CREATE POLICY "System: Allow Bridge Inserts" ON public.gas_logs FOR INSERT WITH CHECK (true);

DROP POLICY IF EXISTS "System: Allow Bridge Incident Inserts" ON public.incidents;
CREATE POLICY "System: Allow Bridge Incident Inserts" ON public.incidents FOR INSERT WITH CHECK (true);

-- 5. REALTIME
-- ==========================================
DO $$
BEGIN
  DROP PUBLICATION IF EXISTS supabase_realtime;
  CREATE PUBLICATION supabase_realtime FOR TABLE public.gas_logs, public.incidents, public.devices;
EXCEPTION WHEN OTHERS THEN
  ALTER PUBLICATION supabase_realtime ADD TABLE public.gas_logs, public.incidents, public.devices;
END $$;
